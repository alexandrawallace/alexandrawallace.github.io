# list循环取值
list1 = [1,2,3,4,5]
list2 = []
list2 = [i for i in list1]
print(list2)
# >>>>> [1,2,3,4,5]
# 错误例子【begin】-----------------------
list2 = list1
list2.append(6)
print(list1)
# >>>>> [1,2,3,4,5,6]
print(list2)
# >>>>> [1,2,3,4,5,6]
# 出现同时赋值【end】-----------------------
# 遍历等长list,  zip用法
for i, j in zip(list1,list2):
	print(i,j)
